Failover Monitoring Query

Delegation Analysis Query

SELECT attributes->>'invoker' as invoker, attributes->>'target' as target, attributes->>'phase' as ooda_phase, AVG((attributes->>'duration_seconds')::float) as avg_duration, COUNT(*) as delegations FROM records WHERE message LIKE 'OODA Delegation%' AND start_timestamp > NOW() - INTERVAL '24 hours' GROUP BY 1, 2, 3 ORDER BY delegations DESC
